## TypeScript

Complete TypeScript examples with types, interfaces, and generics:

```typescript
// Type definitions and interfaces
type User = {
  id: string
  name: string
  email: string
  roles: ("admin" | "editor" | "viewer")[]
}

interface ApiResponse<T> {
  data: T
  status: number
  message?: string
}

// Generics and utility functions
export async function fetchUser(id: string): Promise<ApiResponse<User>> {
  const response = await fetch(`/api/users/${id}`)
  const data = await response.json()
  
  if (!response.ok) {
    throw new Error(data.message ?? "Failed to fetch user")
  }
  
  return {
    data,
    status: response.status,
    message: "User fetched successfully"
  }
}

// Using the function
const result = await fetchUser("user-123")
console.log(`Hello, ${result.data.name}!`)
```

  initialCount?: number
  step?: number
}

      <h2 className="text-2xl font-bold">Count: {count}</h2>

        <Button 
          onClick={() => setCount(c => c + step)}
          variant="default"
        >
          Increment

        <Button 
          onClick={() => setCount(c => c - step)}
          variant="outline"
        >
          Decrement
        </Button>
        
        <Button 
          onClick={() => setCount(initialCount)}
          variant="ghost"
        >
          Reset
        </Button>
      </div>
    </Card>
  )
}

## Python

Python with decorators, type hints, and dataclasses:

```python
# Python class and decorators
from typing import List, Optional
from dataclasses import dataclass

@dataclass
class User:
    id: int
    name: str
    email: str
    roles: List[str]

def authenticate(func):
    """Decorator for authentication"""
    def wrapper(*args, **kwargs):
        if not check_auth():
            raise PermissionError("Not authenticated")
        return func(*args, **kwargs)
    return wrapper

@authenticate
def get_users() -> List[User]:
    """Fetch all users from database"""
    users = [
        User(1, "Alice", "alice@example.com", ["admin"]),
        User(2, "Bob", "bob@example.com", ["editor"]),
    ]
    return users

# List comprehension
admin_users = [u for u in get_users() if "admin" in u.roles]
print(f"Found {len(admin_users)} admin users")
```

## SQL

-- PostgreSQL query example
SELECT 
  u.id,
  u.name,
  u.email,
  COUNT(p.id) as post_count,
  MAX(p.created_at) as last_post_date
FROM users u
LEFT JOIN posts p ON u.id = p.author_id
WHERE u.status = 'active'
  AND p.published = true
GROUP BY u.id, u.name, u.email
HAVING COUNT(p.id) > 5
ORDER BY post_count DESC
LIMIT 10;

CREATE TABLE users (

func main() {
  http.HandleFunc("/user", getUser)
  http.ListenAndServe(":8080", nil)
}

## C++

C++ with templates and STL:

```cpp
#include <iostream>
#include <vector>
#include <string>

struct User {
    int id;
    std::string name;
    std::string email;

    User(int id, const std::string& name, const std::string& email)
        : id(id), name(name), email(email) {}

    void display() const {
        std::cout << "User(" << id << ", " << name << ", " << email << ")\n";
    }
};

template<typename T>
class Repository {
private:
    std::vector<T> items;

public:
    void add(const T& item) {
        items.push_back(item);
    }

    void display_all() const {
        for (const auto& item : items) {
            item.display();
        }
    }
};

int main() {
    Repository<User> repo;
    repo.add(User(1, "Alice", "alice@example.com"));
    repo.add(User(2, "Bob", "bob@example.com"));
    repo.display_all();
    return 0;
}
```
